\pblock{Setup}{%
\textbf{Models (all 7B):} primary within-architecture analysis on \textbf{DeepSeek-Prover-V1.5-RL} paired with non-RL \textbf{V1.5-Base}; cross-model replication on RL-trained \textbf{DeepSeek-Prover-V2-7B} (reasoning-mode) and SFT-trained \textbf{Goedel-Prover}.\\
\textbf{Benchmark:} miniF2F-test (244 Lean~4 / Mathlib theorems).\\
\textbf{Schedule:} 15 hand-picked tactic skeletons (\texttt{simp}, \texttt{intro}, \texttt{constructor}, \texttt{aesop}, \dots) $\times$ 8 NL goal hints = 120-slot grid. Budgets $k \in \{16,32,64\}$ take the first $k$ slots.\\
\textbf{Decoding:} temperature 0.6, top-$p$ 0.95; max 1024 tokens (V1.5 / Goedel completion-mode) or 8192 tokens (V2 reasoning-mode). \textbf{Verifier:} \texttt{lake env lean --json}, 120\,s timeout, 8-way sharding by $i \bmod 8$.
}

\pblock{Mode collapse: A-RL plateaus at $k{=}32$}{%
\begin{center}
\includegraphics[width=\linewidth]{../figures/scaling.pdf}
\end{center}
\vspace{0.2em}
\centering
\renewcommand{\arraystretch}{1.18}
\begin{tabular}{lccc}
\toprule
                              & $k{=}16$       & $k{=}32$       & $k{=}64$       \\ \midrule
A-RL (i.i.d.)                 & 38/244 (15.6\%) & 42/244 (17.2\%) & 42/244 (17.2\%) \\
\textbf{B-RL (skeleton)}      & \textbf{55} (22.5\%) & \textbf{58} (23.8\%) & \textbf{60} (24.6\%) \\
$\Delta$ absolute             & $+17$           & $+16$           & $+18$           \\
$\Delta$ relative             & $+44.7\%$       & $+38.1\%$       & $+42.9\%$       \\
\bottomrule
\end{tabular}\\[0.4em]
\justifying\small\color{muted}
Across-seed: mean $\Delta = +12.3 \pm 4.2$ theorems at $k{=}16$ ($n{=}3$ seeds, sign positive every seed).
}

\pblock{Diversity ablation: structural content drives the gain}{%
\begin{center}
\includegraphics[width=\linewidth]{../figures/ablation.pdf}
\end{center}
\justifying
\textbf{C2 < A-RL = C1 < C3 < B-RL.} Irrelevant tokens hurt; paraphrase ties the baseline; \textbf{C3 (NL hint only, no tactic prefix) alone closes 58\% of the A$\to$B gap}; tactic skeletons go furthest.\\[0.2em]
The RL-collapsed policy doesn't need to be \emph{over-ridden} by a forced first tactic --- just \emph{re-conditioned} on a different prompt prefix, and a different head of its narrow distribution becomes accessible.
}

\pblock{Phenomenon is RL-specific}{%
DeepSeek-Prover-V1.5-Base (same model, no RL fine-tune) proves \textbf{0/244}
theorems at every $k$, with \emph{or} without skeletons.
\begin{itemize}
\setlength{\itemsep}{0.15em}
\item Base outputs \texttt{sorry} or echoes the statement in 58.8\% of attempts.
\item RL model emits \texttt{sorry} in $<\!0.2\%$ of attempts; collapse manifests as failing-but-valid \texttt{rw}/\texttt{apply} sequences.
\end{itemize}
\vspace{0.15em}
\textbf{RL is the locus of both proof capability and the inference-time collapse that limits it.}
}

\pblock{Per-skeleton contribution (within $k{=}16$)}{%
\begin{center}
\includegraphics[width=\linewidth]{../figures/per_skeleton.pdf}
\end{center}
\justifying
24 B-RL $k{=}16$ shards, within-shard counting. Empty skeleton carries most of the load
(3.29/4.54 solves per shard); the other 14 skeletons together pick up the long
tail $-$ exactly the 17 theorems A-RL i.i.d.\ misses.
}

\pblock{Cross-model generalisation}{%
A$\to$B contrast on two further 7B Lean provers:
\begin{itemize}
\setlength{\itemsep}{0.1em}
\item \textbf{RL-trained V2-7B:} $+3$ \emph{frontier} theorems no i.i.d.\ baseline reaches.
\item \textbf{SFT Goedel-Prover:} $\mathbf{-10.0 \pm 4.4}$ theorems ($n{=}3$ seeds). \emph{Helps RL, hurts SFT.}
\end{itemize}
}
